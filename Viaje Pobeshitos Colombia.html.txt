        function updateUI() {
            const plan = tripData[currentPlan];
            totalCostEl.textContent = formatCurrency(plan.totalCost);

                budgetChart.data.datasets[0].data = plan.budget.data;
                budgetChart.update();
            } else {
                renderChart(plan);
            }
            
            updateBudgetDetails(plan);
            createItinerary(plan);
        }

        // Gemini API integration for Packing List
        generatePackingListBtn.addEventListener('click', async () => {
            generatePackingListBtn.disabled = true;
            packingListSpinner.classList.remove('hidden');
            packingListOutput.classList.add('hidden');
            packingListContent.innerHTML = '';

            const currentItinerary = tripData[currentPlan].itinerary;
            // Extract unique destinations and key activities for the prompt
            const destinations = [...new Set(currentItinerary.map(item => {
                if (item.title.includes('Bogot√°')) return 'Bogot√° (clima fr√≠o/templado)';
                if (item.title.includes('Medell√≠n')) return 'Medell√≠n (clima templado)';
                if (item.title.includes('Cartagena') || item.title.includes('Islas del Rosario')) return 'Cartagena (clima c√°lido/playa)';
                if (item.title.includes('Santa Marta') || item.title.includes('Tayrona') || item.title.includes('Barranquilla')) return 'Santa Marta y Barranquilla (clima c√°lido/playa, senderismo)';
                return '';
            }))].filter(Boolean).join(', ');

            const activities = [...new Set(currentItinerary.map(item => {
                if (item.details.includes('senderismo') || item.details.includes('Parque Tayrona')) return 'senderismo';
                if (item.details.includes('playas') || item.details.includes('snorkel') || item.details.includes('catamar√°n')) return 'actividades de playa';
                if (item.details.includes('museos') || item.details.includes('Centro Hist√≥rico') || item.details.includes('cultura')) return 'visitas culturales';
                if (item.details.includes('Comuna 13') || item.details.includes('Guatap√©')) return 'excursiones';
                if (item.details.includes('compras')) return 'compras';
                return '';
            }))].filter(Boolean).join(', ');

            const prompt = `Genera una lista de equipaje para un viaje de 22 d√≠as a Colombia en agosto, visitando ${destinations}. Las actividades incluyen ${activities}. Considera un viaje para dos personas. La respuesta debe ser un JSON con un array de objetos, donde cada objeto tiene una propiedad 'category' (e.g., 'Ropa', 'Documentos', 'Salud') y una propiedad 'items' que es un array de strings con los elementos de esa categor√≠a.`;

            try {
                let chatHistory = [];
                chatHistory.push({ role: "user", parts: [{ text: prompt }] });
                const payload = {
                    contents: chatHistory,
                    generationConfig: {
                        responseMimeType: "application/json",
                        responseSchema: {
                            type: "ARRAY",
                            items: {
                                type: "OBJECT",
                                properties: {
                                    "category": { "type": "STRING" },
                                    "items": {
                                        "type": "ARRAY",
                                        "items": { "type": "STRING" }
                                    }
                                },
                                "propertyOrdering": ["category", "items"]
                            }
                        }
                    }
                };
                const apiKey = ""; // If you want to use models other than gemini-2.0-flash or imagen-3.0-generate-002, provide an API key here. Otherwise, leave this as-is.
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&
                    result.candidates[0].content && result.candidates[0].content.parts &&
                    result.candidates[0].content.parts.length > 0) {
                    const json = result.candidates[0].content.parts[0].text;
                    const parsedJson = JSON.parse(json);

                    parsedJson.forEach(cat => {
                        const categoryDiv = document.createElement('div');
                        categoryDiv.innerHTML = `<h4 class="font-semibold text-lg text-gray-800 mb-2">${cat.category}</h4>`;
                        const ul = document.createElement('ul');
                        ul.className = 'list-disc list-inside ml-4 space-y-1';
                        cat.items.forEach(item => {
                            const li = document.createElement('li');
                            li.textContent = item;
                            ul.appendChild(li);
                        });
                        categoryDiv.appendChild(ul);
                        packingListContent.appendChild(categoryDiv);
                    });
                    packingListOutput.classList.remove('hidden');
                } else {
                    packingListContent.innerHTML = '<p class="text-red-500">Error al generar la lista de equipaje. Int√©ntalo de nuevo.</p>';
                    packingListOutput.classList.remove('hidden');
                }
            } catch (error) {
                console.error('Error calling Gemini API:', error);
                packingListContent.innerHTML = '<p class="text-red-500">Error al conectar con el servicio de IA. Por favor, int√©ntalo m√°s tarde.</p>';
                packingListOutput.classList.remove('hidden');
            } finally {
                generatePackingListBtn.disabled = false;
                packingListSpinner.classList.add('hidden');
            }
        });

        });
